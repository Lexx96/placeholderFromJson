Future<List<UserDbModel>> getUsersFromDb() async {
  List users = await UsersRepository.getAllUsersFromDb();
  if (users.isNotEmpty) {
    return users.map((e) => UserDbModel.fromJson(e as Map<String, dynamic>)).toList();
  }

  return <UserDbModel>[];
}
